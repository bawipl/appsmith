SELECT 
  p.name AS platform,
  p.id AS platform_id,
  COALESCE(SUM(CASE WHEN co.direction = 'in' THEN co.amount ELSE -co.amount END), 0) AS balance
FROM platforms p
LEFT JOIN cashoperations co ON co.platform_id = p.id 
  AND co.owner_id = {{Owner2Select.selectedOptionValue || 'XXXXX'}}
  AND co.currency_id = {{Currency1Select.selectedOptionValue ? Currency1Select.selectedOptionValue : -1}}::INTEGER
GROUP BY p.id, p.name
HAVING COALESCE(SUM(CASE WHEN co.direction = 'in' THEN co.amount ELSE -co.amount END), 0) > 0
ORDER BY balance DESC;
